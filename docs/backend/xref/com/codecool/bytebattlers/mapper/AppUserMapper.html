6   import org.mapstruct.*;
7   
8   import java.util.Set;
9   import java.util.UUID;
10  import java.util.stream.Collectors;
11  
12  @Mapper(unmappedTargetPolicy = ReportingPolicy.IGNORE, componentModel = MappingConstants.ComponentModel.SPRING)
13  public interface AppUserMapper {
14      AppUser toEntity(AppUserDto appUserDto);
15  
16      @AfterMapping
17      default void linkReviews(@MappingTarget AppUser appUser) {
18          appUser.getReviews().forEach(review -> review.setAppUser(appUser));
19      }
20  
21      @Mapping(target = "favoriteBoardGamePublicIDS", expression = "java(favoriteBoardGameToFavoriteBoardGamePublicIDS(appUser.getFavoriteBoardGames()))")
22      AppUserDto toDto(AppUser appUser);
23  
24      @BeanMapping(nullValuePropertyMappingStrategy = NullValuePropertyMappingStrategy.IGNORE)
25      AppUser partialUpdate(AppUserDto appUserDto, @MappingTarget AppUser appUser);
26  
27      default Set<UUID> favoriteBoardGameToFavoriteBoardGamePublicIDS(Set<BoardGame> favoriteBoardGame) {
28          return favoriteBoardGame.stream().map(BoardGame::getPublicID).collect(Collectors.toSet());
29      }
30  }
