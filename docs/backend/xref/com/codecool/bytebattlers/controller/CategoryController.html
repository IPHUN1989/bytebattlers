1   package com.codecool.bytebattlers.controller;
2   
3   import com.codecool.bytebattlers.controller.dto.CategoryDto;
4   import com.codecool.bytebattlers.controller.exception.ResourceNotFoundException;
5   import com.codecool.bytebattlers.service.CategoryService;
6   import org.springframework.beans.factory.annotation.Autowired;
7   import org.springframework.http.HttpStatus;
8   import org.springframework.http.ResponseEntity;
9   import org.springframework.web.bind.annotation.*;
10  
11  import java.util.List;
12  import java.util.UUID;
13  
14  @RestController
15  @RequestMapping("/api/categories")
16  public class CategoryController {
17  
18  
19      private final CategoryService categoryService;
20  
21      @Autowired
22      public CategoryController(CategoryService categoryService) {
23          this.categoryService = categoryService;
24      }
25  
26      @GetMapping
27      public ResponseEntity<List<CategoryDto>> getAllCategory() {
28          if (categoryService.findAll().isEmpty()) {
29              return new ResponseEntity<>(HttpStatus.NO_CONTENT);
30          } else {
31              return new ResponseEntity<>(categoryService.findAll(), HttpStatus.OK);
32          }
33      }
34  
35      @GetMapping("/{id}")
36      public ResponseEntity<CategoryDto> getCategoryById(@PathVariable UUID id) {
37          if (categoryService.findById(id) == null) {
38              throw new ResourceNotFoundException("Not found Tutorial with id = " + id);
39          } else {
40              return new ResponseEntity<>(categoryService.findById(id), HttpStatus.OK);
41          }
42      }
43  
44      @PostMapping
45      public ResponseEntity<CategoryDto> addNewCategory(@RequestBody CategoryDto board) {
46          return new ResponseEntity<>(categoryService.save(board), HttpStatus.CREATED);
47      }
48  
49      @DeleteMapping("/{id}")
50      public ResponseEntity<CategoryDto> deleteCategoryById(@PathVariable UUID id) {
51          categoryService.deleteById(id);
52          return new ResponseEntity<>(HttpStatus.NO_CONTENT);
53      }
54  }
