20  @Service
21  public class BoardGameService {
22  
23      private final BoardGameRepository boardGameRepository;
24      private final PublisherService publisherService;
25      private final BoardGameMapper boardGameMapper;
26      private final CategoryRepository categoryRepository;
27      private final AppUserRepository appUserRepository;
28  
29      private final RatingService ratingService;
30  
31      @Autowired
32      public BoardGameService(
33              BoardGameRepository boardGameRepository,
34              PublisherService publisherService,
35              BoardGameMapper boardGameMapper,
36              CategoryRepository categoryRepository,
37              AppUserRepository appUserRepository, RatingService ratingService) {
38          this.boardGameRepository = boardGameRepository;
39          this.publisherService = publisherService;
40          this.boardGameMapper = boardGameMapper;
41          this.categoryRepository = categoryRepository;
42          this.appUserRepository = appUserRepository;
43          this.ratingService = ratingService;
44      }

46  
47      public List<BoardGameDto> findAll() {
48          return boardGameRepository.findAll().stream().map(boardGameMapper::toDto).toList();
49      }
50  
51      public BoardGameDto save(BoardGameDto dto) {
52          Set<BoardGameDto.CategoryDto1> categoriesInDTO = dto.categories();
53          Set<Category> categoriesInEntity = categoriesInDTO
54                  .stream().map(categoryDto -> categoryRepository.findCategoryByPublicID(categoryDto.publicID()))
55                  .collect(Collectors.toSet());
56          Set<AppUser> favoritedInEntity = dto.appUserPublicIDS().stream()
57                  .map(appUserRepository::findAppUsersByPublicID).collect(Collectors.toSet());
58          Publisher foundPublisher = publisherService.findByPublicId(dto.publisherPublicID());
59          BoardGame boardGame = boardGameMapper.toEntity(dto);
60          boardGame.setUsersWhoFavorited(favoritedInEntity);
61          boardGame.setCategories(categoriesInEntity);
62          boardGame.setPublisher(foundPublisher);
63  
64          try {
65              BoardGame savedBoardGame = boardGameRepository.save(boardGame);
66              return boardGameMapper.toDto(savedBoardGame);
67          } catch (DataIntegrityViolationException e) {
68              throw new RuntimeException(e.getMessage());
69          }
70      }
71  
72      public BoardGame findByPublicID(UUID publicID) {
73          return boardGameRepository.findBoardGameByPublicID(publicID);
74      }
75  
76      public BoardGameDto findByPublicIdToDTO(UUID publicID) {
77          return boardGameMapper.toDto(boardGameRepository.findBoardGameByPublicID(publicID));
78      }
79  
80      public void deleteById(UUID publicID) {
81          boardGameRepository.deleteByPublicID(publicID);
82      }
83  
84      public List<BoardGame> findBoardGamesByGameName(String boardGameName) {
85          return boardGameRepository.findBoardGamesByGameNameContainingIgnoreCase(boardGameName);
86      }
87  
88      public List<BoardGameDto> findBoardGamesDtoByGameName(String boardGameName) {
89          List<BoardGame> searchedBoardgames = findBoardGamesByGameName(boardGameName);
90          return searchedBoardgames.stream()
91                  .map(boardGameMapper::toDto)
92                  .toList();
93      }
94  
95      public List<BoardGameDto> findBoardGamesByPublisherPublicID(UUID publisherID) {
96          Publisher foundPublisher = publisherService.findByPublicId(publisherID);
97          return boardGameRepository.findBoardGamesByPublisher(foundPublisher)
98                  .stream().map(boardGameMapper::toDto)
99                  .toList();
100     }

102     public List<BoardGameDto> findBoardGamesByCategory(UUID publicId) {
103         Category foundCategory = categoryRepository.findCategoryByPublicID(publicId);
104         return boardGameRepository.findBoardGamesByCategoriesContaining(foundCategory)
105                 .stream().map(boardGameMapper::toDto).toList();
106     }
107 
108     public List<BoardGameDto> findBoardGamesByDescription(String description) {
109         return boardGameRepository.findBoardGamesByDescriptionContainsIgnoreCase(description)
110                 .stream().map(boardGameMapper::toDto).toList();
111     }
112 
113     public List<BoardGameDto> sortByName(String order) {
114         if (order.equals("asc")) {
115             return boardGameRepository.findAllByOrderByGameNameAsc()
116                     .stream().map(boardGameMapper::toDto).toList();
117         } else if (order.equals("desc")) {
118             return boardGameRepository.findAllByOrderByGameNameDesc()
119                     .stream().map(boardGameMapper::toDto)
120                     .toList();
121         }
122         return Collections.emptyList();
123     }
124 
125     public List<BoardGameDto> findByLessThanOrEqualsMaxPlayer(int maxPlayer){
126        return boardGameRepository.findAllByMaxPlayerLessThanEqual(maxPlayer)
127                .stream().map(boardGameMapper::toDto)
128                .toList();
129     }
130 
131     public List<BoardGameDto> findByMoreThanOrEqualsMinPlayer(int minPlayer) {
132         return boardGameRepository.findAllByMinPlayerGreaterThanEqual(minPlayer)
133                 .stream().map(boardGameMapper::toDto)
134                 .toList();
135     }
