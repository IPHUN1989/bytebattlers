3   import com.codecool.bytebattlers.controller.dto.BoardGameDto;
4   import com.codecool.bytebattlers.controller.exception.ResourceNotFoundException;
5   import com.codecool.bytebattlers.mapper.BoardGameMapper;

8   import org.springframework.beans.factory.annotation.Autowired;
9   import org.springframework.http.HttpStatus;
10  import org.springframework.http.ResponseEntity;
11  import org.springframework.web.bind.annotation.*;
12  
13  import java.util.List;
14  import java.util.UUID;
15  
16  @RestController
17  @RequestMapping("/api/games")
18  public class BoardGameController {
19  
20      private static final String ERROR_MESSAGE = "No found game";
21  
22  
23      private final BoardGameService boardGameService;
24  
25      private final BoardGameMapper boardGameMapper;
26  
27      @Autowired
28      public BoardGameController(BoardGameService boardGameService,
29                                 BoardGameMapper boardGameMapper) {
30          this.boardGameService = boardGameService;
31          this.boardGameMapper = boardGameMapper;
32      }
33  
34      @GetMapping
35      public ResponseEntity<List<BoardGameDto>> getAllBoardGame() {
36          if (boardGameService.findAll().isEmpty()) {
37              return new ResponseEntity<>(HttpStatus.NO_CONTENT);
38          } else {
39              return new ResponseEntity<>(boardGameService.findAll(), HttpStatus.OK);
40          }
41      }
42  
43      @PostMapping("/myfavorites")
44      public ResponseEntity<List<BoardGameDto>> getFavoriteBoardGamesByUser(@RequestBody List<UUID> uuids) {
45         List<BoardGame> favBoardGames = uuids.stream().map(boardGameService::findByPublicID).toList();
46         List<BoardGameDto> bgDtos = favBoardGames.stream().map(boardGameMapper::toDto).toList();
47          return new ResponseEntity<>(bgDtos, HttpStatus.OK);
48      }
49  
50  
51      @GetMapping("/{id}")
52      public ResponseEntity<BoardGameDto> getBoardGameById(@PathVariable UUID id) {
53          if (boardGameService.findByPublicID(id) == null) {
54              throw new ResourceNotFoundException("Not found Tutorial with id = " + id);
55          } else {
56              return new ResponseEntity<>(boardGameService.findByPublicIdToDTO(id), HttpStatus.OK);
57          }
58      }
59  
60      @PostMapping("/create")
61      public ResponseEntity<?> addNewBoardGame(@RequestBody BoardGameDto board) {
62          try {
63              BoardGameDto savedBoardGame = boardGameService.save(board);
64              return new ResponseEntity<>(savedBoardGame, HttpStatus.CREATED);
65          } catch (RuntimeException e) {
66              return new ResponseEntity<>(null, HttpStatus.CONFLICT);
67          }    }
68  
69      @DeleteMapping("/{id}")
70      public ResponseEntity<BoardGameDto> deleteBoardGameById(@PathVariable UUID id) {
71          boardGameService.deleteById(id);
72          return new ResponseEntity<>(HttpStatus.NO_CONTENT);
73      }
74  
75      @GetMapping("/search")
76      public ResponseEntity<List<BoardGameDto>> findBoardGamesByName(@RequestParam String boardGameName) {
77          if (boardGameService.findBoardGamesDtoByGameName(boardGameName).isEmpty()) {
78              throw new ResourceNotFoundException(ERROR_MESSAGE);
79          } else {
80              return new ResponseEntity<>(boardGameService.findBoardGamesDtoByGameName(boardGameName), HttpStatus.OK);
81          }
82      }
83      @GetMapping("/publisher")
84      public ResponseEntity<List<BoardGameDto>> findBoardGamesByPublisher(@RequestParam UUID publicID) {
85          if (boardGameService.findBoardGamesByPublisherPublicID(publicID).isEmpty()) {
86              throw new ResourceNotFoundException(ERROR_MESSAGE);
87          } else {
88              return new ResponseEntity<>(boardGameService.findBoardGamesByPublisherPublicID(publicID), HttpStatus.OK);
89          }
90      }
91  
92      @GetMapping("/category")
93      public ResponseEntity<List<BoardGameDto>> findBoardGamesByCategory(@RequestParam UUID publicID) {
94          if (boardGameService.findBoardGamesByCategory(publicID).isEmpty()) {
95              throw new ResourceNotFoundException(ERROR_MESSAGE);
96          } else {
97              return new ResponseEntity<>(boardGameService.findBoardGamesByCategory(publicID), HttpStatus.OK);
98          }
99      }
100 
101     @GetMapping("/sort")
102     public ResponseEntity<List<BoardGameDto>> sortBy(@RequestParam String sort) {
103         if (boardGameService.sortByName(sort).isEmpty()) {
104             throw new ResourceNotFoundException(ERROR_MESSAGE);
105         } else {
106             return new ResponseEntity<>(boardGameService.sortByName(sort), HttpStatus.OK);
107         }
108     }
109 
110     @GetMapping("/description")
111     public ResponseEntity<List<BoardGameDto>> findBoardGamesByDescription(@RequestParam String desc) {
112         if (boardGameService.findBoardGamesByDescription(desc).isEmpty()) {
113             throw new ResourceNotFoundException(ERROR_MESSAGE);
114         } else {
115             return new ResponseEntity<>(boardGameService.findBoardGamesByDescription(desc), HttpStatus.OK);
116         }
117     }
118 
119     @GetMapping("/maxplayer")
120     public ResponseEntity<List<BoardGameDto>> findAllBoardGamesLesserThanOrEqualsByMaxPlayer(@RequestParam int max) {
121         if (boardGameService.findByLessThanOrEqualsMaxPlayer(max).isEmpty()) {
122             throw new ResourceNotFoundException(ERROR_MESSAGE);
123         } else {
124             return new ResponseEntity<>(boardGameService.findByLessThanOrEqualsMaxPlayer(max), HttpStatus.OK);
125         }
126     }

128     @GetMapping("/minplayer")
129     public ResponseEntity<List<BoardGameDto>> findAllBoardGamesGreaterThanOrEqualsByMinPlayer(@RequestParam int min) {
130         if (boardGameService.findByMoreThanOrEqualsMinPlayer(min).isEmpty()) {
131             throw new ResourceNotFoundException(ERROR_MESSAGE);
132         } else {
133             return new ResponseEntity<>(boardGameService.findByMoreThanOrEqualsMinPlayer(min), HttpStatus.OK);
