12  import org.springframework.stereotype.Service;
13  
14  import java.util.List;
15  import java.util.UUID;
16  
17  @Service
18  public class RatingService {
19  
20      private final RatingMapper ratingMapper;
21  
22      private final RatingRepository ratingRepository;
23  
24      private final BoardGameRepository boardGameRepository;
25  
26      private final AppUserRepository appUserRepository;
27  
28  
29      @Autowired
30      public RatingService(RatingMapper ratingMapper, RatingRepository ratingRepository, BoardGameRepository boardGameRepository, AppUserRepository appUserRepository) {
31          this.ratingMapper = ratingMapper;
32          this.ratingRepository = ratingRepository;
33          this.boardGameRepository = boardGameRepository;
34          this.appUserRepository = appUserRepository;
35      }
36  
37      public List<RatingDto> findAll() {
38          return ratingRepository.findAll().stream().map(ratingMapper::toDto).toList();
39      }
40  
41      public RatingDto save(RatingDto dto) {
42          BoardGame boardGame = boardGameRepository.findBoardGameByPublicID(dto.boardGamePublicID());
43          AppUser appUser = appUserRepository.findAppUsersByPublicID(dto.appUserPublicID());
44          Rating rating = ratingMapper.toEntity(dto);
45          rating.setAppUser(appUser);
46          rating.setBoardGame(boardGame);
47          ratingRepository.save(rating);
48          return ratingMapper.toDto(rating);
49      }
50  
51      public void delete(Rating entity) {
52          ratingRepository.delete(entity);
53      }
54  
55      public List<Rating> findAllRatingsByBoardGame_Id(UUID boardGameId) {
56          return ratingRepository.findAllByBoardGame_PublicID(boardGameId);
57      }
58  
59      public Rating checkIfRatingExist(UUID appUserPublicID, UUID boardGamePublicID) {
60          BoardGame boardGame = boardGameRepository.findBoardGameByPublicID(boardGamePublicID);
61          AppUser appUser = appUserRepository.findAppUsersByPublicID(appUserPublicID);
62          return ratingRepository.findByAppUserAndAndBoardGame(appUser, boardGame);
63      }
64  }
