41      @Override
42      public final boolean equals(Object o) {
43          if (this == o) return true;
44          if (o == null) return false;
45          Class<?> oEffectiveClass = o instanceof HibernateProxy ? ((HibernateProxy) o).getHibernateLazyInitializer().getPersistentClass() : o.getClass();
46          Class<?> thisEffectiveClass = this instanceof HibernateProxy ? ((HibernateProxy) this).getHibernateLazyInitializer().getPersistentClass() : this.getClass();
47          if (thisEffectiveClass != oEffectiveClass) return false;
48          Rating rating = (Rating) o;
49          return getId() != null && Objects.equals(getId(), rating.getId());
50      }
51
