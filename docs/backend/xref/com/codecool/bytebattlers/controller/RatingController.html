3   import com.codecool.bytebattlers.controller.dto.RatingDto;
4   import com.codecool.bytebattlers.mapper.RatingMapper;
5   import com.codecool.bytebattlers.service.RatingService;
6   import org.springframework.beans.factory.annotation.Autowired;
7   import org.springframework.http.HttpStatus;
8   import org.springframework.http.ResponseEntity;
9   import org.springframework.web.bind.annotation.*;
10  
11  import java.util.List;
12  import java.util.UUID;
13  
14  @RestController
15  @RequestMapping("/api/ratings")
16  public class RatingController {
17  
18      private final RatingService ratingService;
19      private final RatingMapper ratingMapper;
20  
21      @Autowired
22      public RatingController(RatingService ratingService,
23                              RatingMapper ratingMapper) {
24          this.ratingService = ratingService;
25          this.ratingMapper = ratingMapper;
26      }
27  
28      @GetMapping
29      public ResponseEntity<List<RatingDto>> getAllRatings() {
30          if (ratingService.findAll().isEmpty()) {
31              return new ResponseEntity<>(HttpStatus.NO_CONTENT);
32          } else {
33              return new ResponseEntity<>(ratingService.findAll(), HttpStatus.OK);
34          }
35      }
36  
37      @GetMapping("/check-existence")
38      public ResponseEntity<RatingDto> checkRating(@RequestParam(name = "appUserPublicID") UUID appUserPublicID,
39                                                   @RequestParam(name = "boardGamePublicID") UUID boardGamePublicID) {
40          if (ratingService.checkIfRatingExist(appUserPublicID, boardGamePublicID) == null) {
41              return new ResponseEntity<>(HttpStatus.NO_CONTENT);
42          } else {
43              return new ResponseEntity<>(ratingMapper.toDto(ratingService.checkIfRatingExist(appUserPublicID, boardGamePublicID)), HttpStatus.OK);
44          }
45      }
46  
47      @PostMapping
48      public ResponseEntity<RatingDto> addNewRating(@RequestBody RatingDto rating) {
49          return new ResponseEntity<>(ratingService.save(rating), HttpStatus.CREATED);
50      }
51  }
