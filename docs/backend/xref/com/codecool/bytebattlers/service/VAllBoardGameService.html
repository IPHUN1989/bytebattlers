1   package com.codecool.bytebattlers.service;
2   
3   import com.codecool.bytebattlers.controller.dto.VAllBoardGameDto;
4   import com.codecool.bytebattlers.mapper.VAllBoardGameMapper;
5   import com.codecool.bytebattlers.model.VAllBoardGame;
6   import com.codecool.bytebattlers.repository.VAllBoardGameRepository;
7   import org.springframework.beans.factory.annotation.Autowired;
8   import org.springframework.stereotype.Service;
9   
10  import java.util.Collections;
11  import java.util.List;
12  
13  @Service
14  public class VAllBoardGameService {
15  
16      private final VAllBoardGameRepository vAllBoardGameRepository;
17      private final VAllBoardGameMapper vAllBoardGameMapper;
18  
19      @Autowired
20      public VAllBoardGameService(VAllBoardGameRepository vAllBoardGameRepository, VAllBoardGameMapper vAllBoardGameMapper) {
21          this.vAllBoardGameRepository = vAllBoardGameRepository;
22          this.vAllBoardGameMapper = vAllBoardGameMapper;
23      }
24  
25      public List<VAllBoardGameDto> findAll() {
26          return vAllBoardGameRepository.findAll().stream().map(vAllBoardGameMapper::toDto).toList();
27      }
28  
29      public List<VAllBoardGame> findVAllBoardGamesByGameName(String boardGameName) {
30          return vAllBoardGameRepository.findBoardGamesByGameNameContainingIgnoreCase(boardGameName);
31      }
32  
33      public List<VAllBoardGameDto> findVAllBoardGamesDtoByGameName(String boardGameName) {
34          List<VAllBoardGame> searchedBoardgames = findVAllBoardGamesByGameName(boardGameName);
35          return searchedBoardgames.stream()
36                  .map(vAllBoardGameMapper::toDto)
37                  .toList();
38      }
39  
40      public List<VAllBoardGameDto> findAllBoardGamesByPublisher(String publisherName) {
41          return vAllBoardGameRepository.findVAllBoardGamesByPublisherName(publisherName)
42                  .stream().map(vAllBoardGameMapper::toDto)
43                  .toList();
44      }
45  
46      public List<VAllBoardGameDto> findVAllBoardGamesByCategory(String categoryName) {
47          return vAllBoardGameRepository.findVAllBoardGamesByCategoriesContaining(categoryName)
48                  .stream().map(vAllBoardGameMapper::toDto).toList();
49      }
50  
51      public List<VAllBoardGameDto> findVAllBoardGamesByDescription(String description) {
52          return vAllBoardGameRepository.findVAllBoardGamesByDescriptionContainingIgnoreCase(description)
53                  .stream().map(vAllBoardGameMapper::toDto).toList();
54      }
55  
56      public List<VAllBoardGameDto> sortByName(String order) {
57          if (order.equals("asc")) {
58              return vAllBoardGameRepository.findAllByOrderByGameNameAsc()
59                      .stream().map(vAllBoardGameMapper::toDto).toList();
60          } else if (order.equals("desc")) {
61              return vAllBoardGameRepository.findAllByOrderByGameNameDesc()
62                      .stream().map(vAllBoardGameMapper::toDto)
63                      .toList();
64          }
65          return Collections.emptyList();
66      }
67  
68      public List<VAllBoardGameDto> findByLessThanOrEqualsMaxPlayer(int maxPlayer){
69          return vAllBoardGameRepository.findAllByMaxPlayerLessThanEqual(maxPlayer)
70                  .stream().map(vAllBoardGameMapper::toDto)
71                  .toList();
72      }
73  
74      public List<VAllBoardGameDto> findByMoreThanOrEqualsMinPlayer(int minPlayer) {
75          return vAllBoardGameRepository.findAllByMinPlayerGreaterThanEqual(minPlayer)
76                  .stream().map(vAllBoardGameMapper::toDto)
77                  .toList();
78      }
79  
80      public List<VAllBoardGameDto> findByMoreThanOrEqualRating(double rating) {
81          return vAllBoardGameRepository.findAllByAverageRatingGreaterThanEqual(rating)
82                  .stream().map(vAllBoardGameMapper::toDto)
83                  .toList();
84      }

86  
87  }
